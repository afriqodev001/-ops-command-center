      "application": "Auth Service",

          "additional_emails": ["portal-team@x.com"]

          "app": "Mobile App",
          "description": "App start fails for unauthenticated users"
        }
      ],
      "notify_partners": true,
      "notification_emails": ["app-owners@x.com"],
      "suppression": true,
      "suppression_records": ["ALRT-1234", "ALRT-5678"],
      "banner": true,
      "banner_message": "Auth deploy in progress — login may be intermittent"
    }
  ]
}
